\begin{gather*}
  \mathcal{F}\left\{f(t)\right\} \equiv F(\omega) = \frac{1}{\sqrt{2 \pi}}
  \int_{-\infty}^{+\infty} f(t) \mathrm{e}^{-i\omega t} dt \,{dt} \\
  \mathcal{F}^{-1}\left\{F(\omega)\right\} \equiv f(t) = \frac{1}{\sqrt{2 \pi}} 
  \int_{-\infty}^{+\infty} F(\omega) \mathrm{e}^{i \omega t}  \,{d\omega} 
\end{gather*}

    \item $\Gamma (t+1) = t \cdot \Gamma (t), \quad t \in \mathbb{R} $
    \item $\Gamma (n+1) = n \cdot \Gamma (n) = n!, \quad n \in \mathbb{N}$
